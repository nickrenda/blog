viShiny | Interactive visualizations

Built using the Shiny framework for R, viShiny is an interactive web application that demonstrates certain statistical concepts. This is an ongoing project of mine and a tool that I use when teaching introductory statistics. viShiny implements a modular design (inspired by gammaram/intRo) that makes adding new topics quick and easy.

Some fun snippets of code

Here is some Python code to stream tweets in Iowa that mention the names of any of the current presidential candidates. This script also makes use of the yagmail module to send me email notifications every time an arbitrary number of tweets have been collected.

#!/usr/bin/env python
#
# File Name:     stream2.py
# Author:        Evan Pete Walsh
# Contact:       epwalsh@iastate.edu
# Creation Date: 15-09-2015
# Last Modified: Tue Sep 22 17:29:12 2015
# =============================================================================

'''Capture tweets in Iowa about presidential candidates.'''


from tweepy.streaming import StreamListener
from tweepy import OAuthHandler
from tweepy import Stream
from csv import DictReader
from os.path import expanduser
import sys
import json
import yagmail


# Read in twitter authentication data
home = expanduser("~")
with open(home + "/.twitter") as f:
    reader = DictReader(f)
    data = list(reader)

# Read in gmail login data
with open(home + "/.gmail") as f:
    reader = DictReader(f)
    data2 = list(reader)

# Prepare twitter authentication data
access_token = data[0]['access_token']
access_token_secret = data[0]['access_token_secret']
consumer_key = data[0]['consumer_key']
consumer_secret = data[0]['consumer_secret']

# Prepare gmail login data
username = data2[0]['username']
password = data2[0]['password']

keywords_list = ['trump',                   # 'The golden-haired buffoon'
                 'sanders', 'bernie',       # Bernie Sanders
                 'hillary', 'clinton',      # Hillary Clinton
                 'jeb', 'bush',             # Jeb Bush
                 'carson',                  # Ben Carson
                 'christie',                # Chris Christie
                 'cruz',                    # Ted Cruz
                 'fiorina',                 # Carly Fiorina
                 'gilmore',                 # Jim Gilmore
                 'graham',                  # Lindsey Graham
                 'huckabee',                # Mike Huckabee
                 'jindal',                  # Bobby Jindal
                 'kasich',                  # John Kasich
                 'pataki',                  # George Pataki
                 'rand paul', 'randpaul',   # Rand Paul
                 'rubio',                   # Marco Rubio
                 'santorum'                 # Rick Santorum
                 'biden',                   # Joe Biden
                 'cuomo',                   # Andrew Cuomo
                 'gore',                    # Al Gore
                 'kucinich',                # Dennis Kucinich
                 'schweitzer',              # Brian Schweitzer
                 'chafee',                  # Lincoln Chafee
                 'walker',                  # Scott Walker
                 'perry',                   # Rick Perry
                 'webb',                    # Jim Webb
                 'democrat', 'democratic', 'republican', 'gop', 'tea party']


class OutListener(StreamListener):

    tweet_count = 0

    def on_status(self, status):
        if any(x in status.text.lower() for x in keywords_list):
            place = status._json['place']['full_name']
            if 'iowa' in place.lower() or 'ia' in place.lower():
                output = {}
                output['place'] = place
                output['text'] = status.text
                output['hashtags'] = status._json['entities']['hashtags']
                output['retweeted'] = status._json['retweeted']
                output['retweet_count'] = status._json['retweet_count']
                output['favorited'] = status._json['favorited']
                output['favorited_count'] = status._json['favorite_count']
                output['coordinates'] = status._json['coordinates']
                output['time'] = status._json[u'created_at']
                OutListener.tweet_count += 1
                if OutListener.tweet_count % 50 == 0:
                    yag = yagmail.SMTP(username, password)
                    subject = "Tweet count " + str(OutListener.tweet_count)
                    recipient = 'epwalsh@iastate.edu'
                    yag.send(recipient, subject)
                    print json.dumps(output)

    def on_error(self, status_code):
        print >> sys.stderr, 'Error with status code:', status_code
        yag = yagmail.SMTP(username, password)
        yag.send('epwalsh@iastate.edu', str(status_code))
        return True

    def on_timeout(self):
        print >> sys.stderr, 'Timeout...'
        return True


if __name__ == '__main__':
    auth = OAuthHandler(consumer_key, consumer_secret)
    auth.set_access_token(access_token, access_token_secret)
    listener = OutListener()
    stream = Stream(auth, listener)
    stream.filter(locations=[-96.5, 40.5, -90, 43.5])
